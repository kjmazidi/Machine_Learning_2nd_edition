train/test split

set.seed(1234)
i <- sample(1:nrow(df), 0.75*nrow(df), replace=FALSE)
train <- df[i,]
test <- df[-i,]

linear SVM

library(e1071)
svm1 <- svm(Species~., data=train, kernel="linear", cost=10, scale=TRUE)
summary(svm1)
